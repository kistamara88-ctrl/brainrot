    class Karakterek {
        public:
        virtual ~Karakterek(){}
            virtual void mozog() = 0; // absztrakt
            void eletben_van(){}
        protected:
            int szint = 0;
        private:
            int hp = 100;
    };

    class Harcosok : public Karakterek {
        public:
            virtual void harcol() = 0;   // absztrakt
        protected:
            string nev;
        private:
            int jatekosID;
    };

    class Lovagok: public Harcosok {
        public:
                Lovagok(string n) {
                nev = n;
            }

            void mozog() override {
                cout << "Lovával előre vágtat\n" << endl;
            }

            void harcol() override {
                cout << "A lovag támad\n" << endl;
            }
    };

    int main() {
        Karakterek* k = new Lovagok("Lancelot");

        k->mozog();   // polimorfizmus
        k->eletben_van();   // közös metódus

        delete k;
    }
